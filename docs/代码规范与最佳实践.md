## 3. 错误处理最佳实践

### 3.1 C++ 错误处理

```cpp
// 使用异常处理错误
#include <stdexcept>
#include <system_error>

// 自定义异常类
class MediaException : public std::runtime_error {
public:
    explicit MediaException(const std::string& message)
        : std::runtime_error(message) {}
    
    MediaException(const std::string& message, int errorCode)
        : std::runtime_error(message), m_errorCode(errorCode) {}
    
    int errorCode() const { return m_errorCode; }
    
private:
    int m_errorCode = 0;
};

class DeviceNotFoundException : public MediaException {
public:
    explicit DeviceNotFoundException(const std::string& deviceName)
        : MediaException("Device not found: " + deviceName, 1001) {}
};

// 使用异常
VideoDevice* openDevice(const std::string& deviceId) {
    auto devices = enumerateDevices();
    auto it = std::find_if(devices.begin(), devices.end(),
        [&](const auto& d) { return d.id() == deviceId; });
    
    if (it == devices.end()) {
        throw DeviceNotFoundException(deviceId);
    }
    
    return new VideoDevice(*it);
}

// 捕获异常
try {
    auto device = openDevice("camera-01");
    device->startCapture();
} catch (const DeviceNotFoundException& e) {
    qWarning() << "Device error:" << e.what();
    showUserMessage("Camera not found. Please check device connection.");
} catch (const MediaException& e) {
    qWarning() << "Media error:" << e.what();
    showUserMessage("Failed to open camera. Error code: " + e.errorCode());
} catch (const std::exception& e) {
    qCritical() << "Unexpected error:" << e.what();
    showUserMessage("An unexpected error occurred.");
}

// 使用 Result 类型（函数式错误处理）
template<typename T>
class Result {
public:
    static Result<T> success(T value) {
        return Result(std::move(value), {});
    }
    
    static Result<T> failure(const std::string& error) {
        return Result({}, error);
    }
    
    bool isOk() const { return m_error.empty(); }
    bool isErr() const { return !m_error.empty(); }
    
    const T& unwrap() const {
        if (isErr()) throw std::runtime_error(m_error);
        return m_value;
    }
    
    const std::string& error() const { return m_error; }
    
private:
    Result(T value, std::string error)
        : m_value(std::move(value)), m_error(std::move(error)) {}
    
    T m_value;
    std::string m_error;
};

// 使用 Result
Result<VideoDevice*> openDevice(const std::string& deviceId) {
    auto devices = enumerateDevices();
    auto it = std::find_if(devices.begin(), devices.end(),
        [&](const auto& d) { return d.id() == deviceId; });
    
    if (it == devices.end()) {
        return Result<VideoDevice*>::failure("Device not found");
    }
    
    return Result<VideoDevice*>::success(new VideoDevice(*it));
}

// 调用
auto result = openDevice("camera-01");
if (result.isOk()) {
    auto device = result.unwrap();
    device->startCapture();
} else {
    qWarning() << "Error:" << result.error();
}
```

```cpp
// RAII 资源管理
class FileHandle {
public:
    explicit FileHandle(const std::string& path) {
        m_file = fopen(path.c_str(), "r");
        if (!m_file) {
            throw std::runtime_error("Failed to open file");
        }
    }
    
    ~FileHandle() {
        if (m_file) {
            fclose(m_file);
        }
    }
    
    // 禁止拷贝
    FileHandle(const FileHandle&) = delete;
    FileHandle& operator=(const FileHandle&) = delete;
    
    // 允许移动
    FileHandle(FileHandle&& other) noexcept : m_file(other.m_file) {
        other.m_file = nullptr;
    }
    
    FileHandle& operator=(FileHandle&& other) noexcept {
        if (this != &other) {
            if (m_file) fclose(m_file);
            m_file = other.m_file;
            other.m_file = nullptr;
        }
        return *this;
    }
    
    FILE* get() const { return m_file; }
    
private:
    FILE* m_file;
};
```

### 5.3 统计数据采集

```cpp
// WebRTC 统计采集
class StatsCollector : public webrtc::RTCStatsCollectorCallback {
public:
    void OnStatsDelivered(
        const rtc::scoped_refptr<const webrtc::RTCStatsReport>& report) override
    {
        // 处理统计数据
        for (const auto& stats : *report) {
            if (stats.type() == webrtc::RTCInboundRTPStreamStats::kType) {
                auto& inbound = 
                    static_cast<const webrtc::RTCInboundRTPStreamStats&>(stats);
                
                uint64_t bytesReceived = *inbound.bytes_received();
                uint64_t packetsLost = *inbound.packets_lost();
                double jitter = *inbound.jitter();
                
                emit statsReceived(bytesReceived, packetsLost, jitter);
            }
            
            if (stats.type() == webrtc::RTCRemoteOutboundRtpStreamStats::kType) {
                auto& remoteOutbound =
                    static_cast<const webrtc::RTCRemoteOutboundRtpStreamStats&>(stats);
                
                double rtt = *remoteOutbound.round_trip_time();
                emit rttMeasured(rtt * 1000);  // 转换为毫秒
            }
        }
    }
    
signals:
    void statsReceived(uint64_t bytes, uint64_t lost, double jitter);
    void rttMeasured(double rttMs);
};

// 定期采集统计
void startStatsCollection(webrtc::PeerConnectionInterface* pc) {
    QTimer* timer = new QTimer();
    QObject::connect(timer, &QTimer::timeout, [pc]() {
        auto callback = rtc::make_ref_counted<StatsCollector>();
        pc->GetStats(callback.get());
    });
    timer->start(1000);  // 每秒采集一次
}
```

```yaml
  build:
    runs-on: ubuntu-latest
    
    steps:
    - uses: actions/checkout@v3
    
    - name: Install dependencies
      run: |
        sudo apt-get update
        sudo apt-get install -y cmake ninja-build
    
    - name: Configure

    - name: Build
```

## 8. 总结

### 规范要点速查

| 类别 | 规范 |
|------|------|
| **命名** | 类 PascalCase, 函数 camelCase, 变量 m_ 前缀 |
| **格式** | 4 空格缩进, 120 字符行宽 |
| **注释** | 文件头、类、公共方法必须注释 |
| **提交** | type(scope): subject 格式 |
| **错误处理** | 使用异常或 Result 类型 |
| **资源管理** | RAII, 智能指针 |

### 最佳实践要点

1. **代码可读性优先**
2. **一致的命名风格**
3. **完善的错误处理**
4. **资源自动管理**
5. **充分的测试覆盖**
6. **清晰的提交历史**
