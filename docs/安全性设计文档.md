### 1.2 威胁分类

| 威胁类型 | 攻击方式 | 影响 | 风险等级 |
|----------|----------|------|----------|
| **窃听** | 截获音视频数据、信令消息 | 隐私泄露 | 高 |
| **篡改** | 修改传输中的数据 | 数据完整性破坏 | 高 |
| **重放攻击** | 重新发送截获的消息 | 身份欺骗 | 中 |
| **中间人攻击** | 拦截并伪造通信 | 全面安全威胁 | 高 |
| **DDoS 攻击** | 大量请求消耗资源 | 服务不可用 | 中 |
| **注入攻击** | 注入恶意代码或数据 | 系统被控制 | 高 |
| **身份欺骗** | 伪造用户身份 | 未授权访问 | 高 |

### 3.2 JWT Token 设计

```javascript
// JWT Token 结构
const token = {
    // Header
    header: {
        alg: 'RS256',  // RSA 签名
        typ: 'JWT'
    },
    
    // Payload
    payload: {
        // 标准声明
        iss: 'minimeeting.com',      // 签发者
        sub: 'user_123',             // 主题（用户 ID）
        aud: 'signal.minimeeting.com', // 受众
        exp: Math.floor(Date.now() / 1000) + 3600, // 过期时间（1小时）
        iat: Math.floor(Date.now() / 1000),  // 签发时间
        jti: generateUUID(),         // JWT ID
        
        // 自定义声明
        name: 'John Doe',
        role: 'participant',
        permissions: ['join_room', 'create_room']
    },
    
    // Signature
    signature: RSASHA256(
        base64UrlEncode(header) + '.' + base64UrlEncode(payload),
        privateKey
    )
};

// Token 验证
function verifyToken(token, publicKey) {
    const decoded = jwt.verify(token, publicKey, {
        algorithms: ['RS256'],
        issuer: 'minimeeting.com',
        audience: 'signal.minimeeting.com'
    });
    
    // 检查过期
    if (decoded.exp < Date.now() / 1000) {
        throw new Error('Token expired');
    }
    
    // 检查权限
    if (!decoded.permissions.includes('join_room')) {
        throw new Error('Insufficient permissions');
    }
    
    return decoded;
}
```

### 3.3 TURN 认证

```javascript
// TURN 动态凭证生成
function generateTurnCredentials(username, secret, ttl = 86400) {
    const timestamp = Math.floor(Date.now() / 1000) + ttl;
    const turnUsername = `${timestamp}:${username}`;
    
    // HMAC-SHA1 签名
    const hmac = crypto.createHmac('sha1', secret);
    hmac.update(turnUsername);
    const password = hmac.digest('base64');
    
    return {
        username: turnUsername,
        password: password,
        ttl: ttl,
        uris: [
            'turn:turn.minimeeting.com:3478',
            'turn:turn.minimeeting.com:3478?transport=tcp',
            'turns:turn.minimeeting.com:5349'
        ]
    };
}
```

### 3.4 房间访问控制

```javascript
// 房间访问控制
class RoomAccessControl {
    constructor() {
        this.roomPermissions = new Map();
    }
    
    // 创建房间
    createRoom(roomId, creatorId, options = {}) {
        const room = {
            id: roomId,
            creator: creatorId,
            created: Date.now(),
            
            // 访问控制
            isPublic: options.isPublic ?? false,
            password: options.password ? hashPassword(options.password) : null,
            maxParticipants: options.maxParticipants ?? 6,
            allowedUsers: options.allowedUsers ?? [],
            
            // 当前参与者
            participants: new Set()
        };
        
        this.roomPermissions.set(roomId, room);
        return room;
    }
    
    // 加入房间验证
    canJoinRoom(roomId, userId, password = null) {
        const room = this.roomPermissions.get(roomId);
        
        if (!room) {
            return { allowed: false, reason: 'Room not found' };
        }
        
        // 检查房间是否已满
        if (room.participants.size >= room.maxParticipants) {
            return { allowed: false, reason: 'Room is full' };
        }
        
        // 公开房间
        if (room.isPublic) {
            return { allowed: true };
        }
        
        // 密码保护
        if (room.password) {
            if (!password || !verifyPassword(password, room.password)) {
                return { allowed: false, reason: 'Invalid password' };
            }
        }
        
        // 白名单
        if (room.allowedUsers.length > 0 && !room.allowedUsers.includes(userId)) {
            return { allowed: false, reason: 'Not authorized' };
        }
        
        return { allowed: true };
    }
    
    // 加入房间
    joinRoom(roomId, userId) {
        const room = this.roomPermissions.get(roomId);
        if (room) {
            room.participants.add(userId);
        }
    }
    
    // 离开房间
    leaveRoom(roomId, userId) {
        const room = this.roomPermissions.get(roomId);
        if (room) {
            room.participants.delete(userId);
            
            // 创建者离开，销毁房间
            if (userId === room.creator) {
                this.roomPermissions.delete(roomId);
            }
        }
    }
}
```

### 4.2 日志安全

```javascript
// 安全日志配置
class SecureLogger {
    constructor() {
        this.sensitiveFields = ['password', 'token', 'ssrc', 'ip'];
    }
    
    // 脱敏处理
    sanitize(data) {
        const sanitized = { ...data };
        
        for (const field of this.sensitiveFields) {
            if (sanitized[field]) {
                // 部分隐藏
                if (typeof sanitized[field] === 'string') {
                    sanitized[field] = this.maskString(sanitized[field]);
                } else {
                    sanitized[field] = '[REDACTED]';
                }
            }
        }
        
        return sanitized;
    }
    
    // 字符串脱敏
    maskString(str) {
        if (str.length <= 8) {
            return '****';
        }
        return str.substring(0, 4) + '****' + str.substring(str.length - 4);
    }
    
    // 安全日志
    log(level, message, data = {}) {
        const sanitizedData = this.sanitize(data);
        const logEntry = {
            timestamp: new Date().toISOString(),
            level,
            message,
            data: sanitizedData
        };
        
        // 写入日志（不包含敏感信息）
        console.log(JSON.stringify(logEntry));
    }
    
    // 不记录敏感日志
    debug(message, data) {
        this.log('debug', message, data);
    }
    
    info(message, data) {
        this.log('info', message, data);
    }
    
    warn(message, data) {
        this.log('warn', message, data);
    }
    
    error(message, data) {
        this.log('error', message, data);
    }
}

// 使用示例
const logger = new SecureLogger();

    token: 'eyJhbGciOiJIUzI1NiIs...'  // 自动脱敏
});
// 输出: {"timestamp":"...","level":"info","message":"User joined room","data":{"userId":"user_123","roomId":"room_456","token":"eyJh****NiIs..."}}
```

```cpp
// C++ 内存安全实践
class SecureBuffer {
public:
    SecureBuffer(size_t size) : m_size(size) {
        m_data = new uint8_t[size];
    }
    
    ~SecureBuffer() {
        // 安全清除内存
        memset_s(m_data, m_size, 0, m_size);
        delete[] m_data;
        m_data = nullptr;
    }
    
    // 禁止拷贝
    SecureBuffer(const SecureBuffer&) = delete;
    SecureBuffer& operator=(const SecureBuffer&) = delete;
    
    // 移动语义
    SecureBuffer(SecureBuffer&& other) noexcept 
        : m_data(other.m_data), m_size(other.m_size) {
        other.m_data = nullptr;
        other.m_size = 0;
    }
    
    uint8_t* data() { return m_data; }
    size_t size() const { return m_size; }
    
private:
    uint8_t* m_data;
    size_t m_size;
};
```

```javascript
// 应用层 DDoS 防护
class DDoSProtection {
    constructor() {
        this.connectionLimits = new Map();
        this.requestLimits = new Map();
    }
    
    // 连接限制
    checkConnectionLimit(ip) {
        const limit = 10;  // 每个 IP 最大 10 个连接
        const current = this.connectionLimits.get(ip) || 0;
        
        if (current >= limit) {
            return false;
        }
        
        this.connectionLimits.set(ip, current + 1);
        return true;
    }
    
    // 请求速率限制
    checkRequestRate(ip) {
        const maxRequests = 100;  // 每分钟最大请求数
        const windowMs = 60000;
        const now = Date.now();
        
        if (!this.requestLimits.has(ip)) {
            this.requestLimits.set(ip, []);
        }
        
        const requests = this.requestLimits.get(ip);
        
        // 清理过期请求
        while (requests.length && requests[0] < now - windowMs) {
            requests.shift();
        }
        
        if (requests.length >= maxRequests) {
            return false;
        }
        
        requests.push(now);
        return true;
    }
    
    // 清理过期记录
    cleanup() {
        const now = Date.now();
        const windowMs = 60000;
        
        for (const [ip, requests] of this.requestLimits) {
            while (requests.length && requests[0] < now - windowMs) {
                requests.shift();
            }
            if (requests.length === 0) {
                this.requestLimits.delete(ip);
            }
        }
    }
}

// 中间件
const ddosProtection = new DDoSProtection();

app.use((req, res, next) => {

    }

});
```

```javascript
// 输入验证工具
class InputValidator {
    // 房间 ID 验证
    static validateRoomId(roomId) {
        if (!roomId || typeof roomId !== 'string') {
            return { valid: false, error: 'Invalid room ID' };
        }
        
        // 长度限制
        if (roomId.length < 4 || roomId.length > 32) {
            return { valid: false, error: 'Room ID must be 4-32 characters' };
        }
        
        // 字符限制
        if (!/^[a-zA-Z0-9_-]+$/.test(roomId)) {
            return { valid: false, error: 'Invalid characters in room ID' };
        }
        
        return { valid: true };
    }
    
    // 用户名验证
    static validateUsername(username) {
        if (!username || typeof username !== 'string') {
            return { valid: false, error: 'Invalid username' };
        }
        
        if (username.length < 2 || username.length > 32) {
            return { valid: false, error: 'Username must be 2-32 characters' };
        }
        
        // 防止 XSS
        const sanitized = username.replace(/<[^>]*>/g, '');
        if (sanitized !== username) {
            return { valid: false, error: 'Invalid characters in username' };
        }
        
        return { valid: true, sanitized };
    }
    
    // SDP 验证
    static validateSDP(sdp) {
        if (!sdp || typeof sdp !== 'string') {
            return { valid: false, error: 'Invalid SDP' };
        }
        
        // 大小限制
        if (sdp.length > 65536) {  // 64KB
            return { valid: false, error: 'SDP too large' };
        }
        
        // 基本格式检查
        if (!sdp.includes('v=0') || !sdp.includes('o=')) {
            return { valid: false, error: 'Invalid SDP format' };
        }
        
        return { valid: true };
    }
    
    // ICE 候选验证
    static validateICECandidate(candidate) {
        if (!candidate || typeof candidate !== 'object') {
            return { valid: false, error: 'Invalid ICE candidate' };
        }
        
        const required = ['candidate', 'sdpMid', 'sdpMLineIndex'];
        for (const field of required) {
            if (candidate[field] === undefined) {
                return { valid: false, error: `Missing ${field}` };
            }
        }
        
        return { valid: true };
    }
}
```

```javascript
// 安全应急响应
class SecurityIncidentResponse {
    constructor() {
        this.severityLevels = {
            CRITICAL: 1,  // 数据泄露、系统被控
            HIGH: 2,      // 未授权访问、DDoS
            MEDIUM: 3,    // 异常行为、策略违反
            LOW: 4        // 潜在风险、安全告警
        };
    }
    
    // 处理安全事件
    async handleIncident(incident) {
        const severity = this.severityLevels[incident.severity];
        
        // 记录事件
        await this.logIncident(incident);
        
        // 根据严重程度响应
        switch (severity) {
            case 1: // CRITICAL
                await this.handleCriticalIncident(incident);
                break;
            case 2: // HIGH
                await this.handleHighIncident(incident);
                break;
            case 3: // MEDIUM
                await this.handleMediumIncident(incident);
                break;
            case 4: // LOW
                await this.handleLowIncident(incident);
                break;
        }
    }
    
    // 严重事件处理
    async handleCriticalIncident(incident) {
        // 1. 立即通知安全团队
        await this.notifySecurityTeam(incident);
        
        // 2. 隔离受影响系统
        await this.isolateSystem(incident.affectedSystem);
        
        // 3. 吊销所有会话
        await this.revokeAllSessions();
        
        // 4. 准备公开声明（如需要）
        if (incident.dataBreach) {
            await this.preparePublicDisclosure(incident);
        }
    }
    
    // 高危事件处理
    async handleHighIncident(incident) {
        // 阻止攻击来源
        await this.blockSource(incident.sourceIP);
        
        // 增加监控
        await this.enhanceMonitoring(incident.type);
    }
}
```
