## 1. 性能指标体系

```cpp
class NetworkQualityDetector {
public:
    enum class NetworkQuality {
        Excellent,  // 丢包 < 2%, RTT < 100ms
        Good,       // 丢包 < 5%, RTT < 200ms
        Fair,       // 丢包 < 10%, RTT < 300ms

    NetworkQuality detect(double packetLoss, double rtt, double jitter) {
        m_packetLoss = packetLoss;
        m_rtt = rtt;
        m_jitter = jitter;
        
        if (packetLoss < 0.02 && rtt < 100) {
            return NetworkQuality::Excellent;
        } else if (packetLoss < 0.05 && rtt < 200) {
            return NetworkQuality::Good;
        } else if (packetLoss < 0.10 && rtt < 300) {
            return NetworkQuality::Fair;
        } else {
            return NetworkQuality::Poor;
        }
    }
    
private:
    double m_packetLoss = 0;
    double m_rtt = 0;
    double m_jitter = 0;
};
```

#### 码率调整策略

```cpp
// GCC 参数配置
struct GCCConfig {
    // 最小/最大码率
    int minBitrate = 50;    // kbps
    int maxBitrate = 2000;  // kbps
    
    // 初始码率
    int startBitrate = 300; // kbps
    
    // 丢包阈值
    double lowLossThreshold = 0.02;   // 2%
    double highLossThreshold = 0.10;  // 10%
    
    // 调整因子
    double increaseFactor = 1.05;     // 加性增
    double decreaseFactor = 0.85;     // 乘性减
};

// 码率调整逻辑
void adjustBitrateBasedOnLoss(double packetLoss, int currentBitrate) {
    if (packetLoss < lowLossThreshold) {
        // 低丢包：增加码率
        int newBitrate = currentBitrate * increaseFactor;
        encoder->setTargetBitrate(std::min(newBitrate, maxBitrate));
    } else if (packetLoss > highLossThreshold) {
        // 高丢包：降低码率
        int newBitrate = currentBitrate * decreaseFactor;
        encoder->setTargetBitrate(std::max(newBitrate, minBitrate));
    }
    // 中等丢包：保持当前码率
}
```

```cpp
public:
```

```cpp
        if (isKeyFrameLost(sequenceNumber)) {
            // 关键帧丢失：请求新关键帧
            requestKeyFrame();
        } else {
            // 非关键帧：尝试帧内插值
            interpolateFrame(sequenceNumber);
        }
    }
    
private:
    void requestKeyFrame() {
        // 发送 PLI (Picture Loss Indication)
        rtcpSender->sendPLI(ssrc);
    }
    
    void interpolateFrame(int seqNum) {
        // 使用前后帧进行插值
        Frame* prev = frameBuffer->getFrame(seqNum - 1);
        Frame* next = frameBuffer->getFrame(seqNum + 1);
        if (prev && next) {
            Frame* interpolated = interpolate(prev, next);
            decoder->decode(interpolated);
        }
    }
};
```

### 4.1 带宽估算

```cpp
// 带宽估算（基于延迟梯度）
class BandwidthEstimator {
public:
    void update(double arrivalTime, double sendTime, int payloadSize) {
        // 计算单向延迟
        double delay = arrivalTime - sendTime;
        
        // 更新延迟梯度
        double delayGradient = delay - m_lastDelay;
        m_lastDelay = delay;
        
        // 趋势检测
        m_trendLine.update(delayGradient);
        
        // 判断拥塞状态
        if (m_trendLine.getTrend() > threshold) {
            // 网络拥塞：降低带宽估算
            m_estimate *= 0.85;
        } else {
            // 网络空闲：增加带宽估算
            m_estimate *= 1.05;
        }
        
        // 限制范围
        m_estimate = std::max(m_minBitrate, std::min(m_maxBitrate, m_estimate));
    }
    
    int getEstimate() const { return m_estimate; }
    
private:
    int m_estimate = 300;  // kbps
    int m_minBitrate = 50;
    int m_maxBitrate = 2000;
    double m_lastDelay = 0;
    TrendLine m_trendLine;
};
```

### 4.2 带宽分配策略

```cpp
// 多流带宽分配
class BandwidthAllocator {
public:
    void allocate(int totalBandwidth, const QList<MediaStream>& streams) {
        // 优先级分配
        int audioBandwidth = allocateAudio(streams.audio);
        int remaining = totalBandwidth - audioBandwidth;
        
        // 视频带宽分配
        int videoCount = streams.videos.size();
        if (videoCount == 1) {
            // 单视频：全部分配
            streams.videos[0].setBitrate(remaining);
        } else {
            // 多视频：按优先级分配
            for (auto& video : streams.videos) {
                int share = remaining * video.priority / totalPriority;
                video.setBitrate(share);
            }
        }
        
        // 屏幕共享高优先级
        if (streams.screenShare) {
            streams.screenShare->setBitrate(remaining * 0.6);
            // 其他视频降级
            for (auto& video : streams.videos) {
                video.setBitrate(video.bitrate() * 0.4);
            }
        }
    }
    
private:
    int allocateAudio(const AudioStream& audio) {
        // 音频固定码率
        switch (audio.quality) {
            case AudioQuality::High:   return 64;   // kbps
            case AudioQuality::Medium: return 32;
            case AudioQuality::Low:    return 16;
        }
    }
};
```

```cpp
// 对象池复用
template<typename T>
class ObjectPool {
public:
    std::shared_ptr<T> acquire() {
        std::lock_guard<std::mutex> lock(m_mutex);
        
        if (m_pool.empty()) {
            return std::make_shared<T>();
        }
        
        auto obj = m_pool.back();
        m_pool.pop_back();
        return obj;
    }
    
    void release(std::shared_ptr<T> obj) {
        std::lock_guard<std::mutex> lock(m_mutex);
        if (m_pool.size() < m_maxSize) {
            obj->reset();  // 重置对象状态
            m_pool.push_back(obj);
        }
    }
    
private:
    std::vector<std::shared_ptr<T>> m_pool;
    std::mutex m_mutex;
    size_t m_maxSize = 100;
};

// 使用对象池

auto frame = framePool.acquire();
// ... 使用 frame
```

```cpp
        // 使用纹理缓存
        glGenTextures(kMaxTextures, m_textures);
        
        // 启用 VSync（可选）
        QSurfaceFormat format;
        format.setSwapInterval(1);  // VSync
    }
    
    void render(const VideoFrame& frame) {
        // 避免频繁的 OpenGL 状态切换
        glUseProgram(m_shaderProgram);
        
        // 直接使用 GPU 纹理
        glBindTexture(GL_TEXTURE_2D, m_textures[frame.textureId()]);
        glTexSubImage2D(GL_TEXTURE_2D, 0, 0, 0, 
                       frame.width(), frame.height(),
                       GL_RGBA, GL_UNSIGNED_BYTE, frame.data());
        
        // 批量渲染
        glDrawArrays(GL_TRIANGLE_STRIP, 0, 4);
    }
};
```
