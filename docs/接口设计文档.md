### 2.1 信令服务器接口

#### 2.1.1 SignalingServer

信令服务器主类。

```javascript
class SignalingServer {
    constructor(port) {
        this.port = port;
        this.wss = null;
        this.onlineUsers = new Map();      // userId -> { ws, roomId }
        this.rooms = new Map();            // roomId -> { roomName, users: Set }
        this.heartbeatTimers = new Map();  // userId -> timer
    }

    start() {
        // 启动服务器
    }

    stop() {
        // 停止服务器
    }

    // 消息处理
    handleMessage(ws, message) {
        const data = JSON.parse(message);
        switch (data.type) {
            case 'login':
                this.handleLogin(ws, data);
                break;
            case 'createRoom':
                this.handleCreateRoom(ws, data);
                break;
            case 'joinRoom':
                this.handleJoinRoom(ws, data);
                break;
            case 'leaveRoom':
                this.handleLeaveRoom(ws, data);
                break;
            case 'offer':
            case 'answer':
            case 'iceCandidate':
                this.handleSignaling(ws, data);
                break;
            case 'heartbeat':
                this.handleHeartbeat(ws, data);
                break;
        }
    }

    // 用户登录
    handleLogin(ws, data) {}

    // 创建房间
    handleCreateRoom(ws, data) {}

    // 加入房间
    handleJoinRoom(ws, data) {}

    // 离开房间
    handleLeaveRoom(ws, data) {}

    // 信令转发
    handleSignaling(ws, data) {}

    // 心跳处理
    handleHeartbeat(ws, data) {}

    // 发送消息
    sendMessage(ws, data) {
        ws.send(JSON.stringify(data));
    }

    // 广播房间消息
    broadcastToRoom(roomId, data, excludeUserId = null) {}

    // 清理断开用户
    cleanupUser(userId) {}
}
```

#### 2.1.2 RoomManager

房间管理器。

```javascript
class RoomManager {
    constructor() {
        this.rooms = new Map();  // roomId -> Room
    }

    createRoom(roomName, creatorId) {
        const roomId = generateRoomId();
        const room = {
            id: roomId,
            name: roomName,
            creator: creatorId,
            users: new Set([creatorId]),
            createdAt: Date.now()
        };
        this.rooms.set(roomId, room);
        return room;
    }

    getRoom(roomId) {
        return this.rooms.get(roomId);
    }

    joinRoom(roomId, userId) {
        const room = this.rooms.get(roomId);
        if (room) {
            room.users.add(userId);
            return true;
        }
        return false;
    }

    leaveRoom(roomId, userId) {
        const room = this.rooms.get(roomId);
        if (room) {
            room.users.delete(userId);
            if (room.users.size === 0) {
                this.rooms.delete(roomId);
            }
            return true;
        }
        return false;
    }

    getRoomUsers(roomId) {
        const room = this.rooms.get(roomId);
        return room ? Array.from(room.users) : [];
    }

    getAllRooms() {
        return Array.from(this.rooms.values());
    }

    deleteRoom(roomId) {
        this.rooms.delete(roomId);
    }
}
```

#### 2.1.3 UserManager

用户管理器。

```javascript
class UserManager {
    constructor() {
        this.users = new Map();  // userId -> User
    }

    addUser(userId, ws) {
        this.users.set(userId, {
            id: userId,
            ws: ws,
            roomId: null,
            lastHeartbeat: Date.now()
        });
    }

    removeUser(userId) {
        this.users.delete(userId);
    }

    getUser(userId) {
        return this.users.get(userId);
    }

    setUserRoom(userId, roomId) {
        const user = this.users.get(userId);
        if (user) {
            user.roomId = roomId;
        }
    }

    updateHeartbeat(userId) {
        const user = this.users.get(userId);
        if (user) {
            user.lastHeartbeat = Date.now();
        }
    }

    isUserOnline(userId) {
        return this.users.has(userId);
    }

    getOnlineUsers() {
        return Array.from(this.users.keys());
    }
}
```

## 3. 回调事件定义

### 3.1 客户端回调事件

| 事件名 | 参数 | 说明 |
|--------|------|------|
| onConnected | - | 信令服务器连接成功 |
| onDisconnected | - | 信令服务器断开连接 |
| onLoginSuccess | userId | 登录成功 |
| onRoomCreated | roomId, roomName | 房间创建成功 |
| onJoinRoomSuccess | roomId, users[] | 加入房间成功 |
| onUserJoined | roomId, userId | 新用户加入房间 |
| onUserLeft | roomId, userId | 用户离开房间 |
| onOffer | from, sdp | 收到 Offer SDP |
| onAnswer | from, sdp | 收到 Answer SDP |
| onIceCandidate | from, candidate, sdpMid, sdpMlineIndex | 收到 ICE Candidate |
| onConnectionStateChange | peerId, state | PeerConnection 状态变化 |
| onError | message | 发生错误 |

### 3.2 服务器回调事件

| 事件名 | 参数 | 说明 |
|--------|------|------|
| onUserConnect | ws | 用户 WebSocket 连接 |
| onUserDisconnect | userId | 用户断开连接 |
| onMessage | ws, message | 收到消息 |
| onError | ws, error | 发生错误 |
