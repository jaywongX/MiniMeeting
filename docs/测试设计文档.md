const { RoomManager } = require('../../src/roomManager');

describe('RoomManager', () => {
    let roomManager;

    beforeEach(() => {
        roomManager = new RoomManager();
    });

    describe('createRoom', () => {
        it('should create a new room', () => {
            const room = roomManager.createRoom('Test Room', 'user_001');
            
            expect(room.roomName).toBe('Test Room');
            expect(room.creatorId).toBe('user_001');
            expect(room.users.has('user_001')).toBe(true);
        });

        it('should generate unique room IDs', () => {
            const room1 = roomManager.createRoom('Room 1', 'user_001');
            const room2 = roomManager.createRoom('Room 2', 'user_002');
            
            expect(room1.roomId).not.toBe(room2.roomId);
        });
    });

    describe('joinRoom', () => {
        it('should add user to room', () => {
            const room = roomManager.createRoom('Test Room', 'user_001');
            const result = roomManager.joinRoom(room.roomId, 'user_002');
            
            expect(result).toBe(true);
            expect(roomManager.getRoomUsers(room.roomId)).toContain('user_002');
        });

        it('should fail if room does not exist', () => {
            const result = roomManager.joinRoom('nonexistent', 'user_001');
            expect(result).toBe(false);
        });

        it('should fail if room is full', () => {
            const room = roomManager.createRoom('Test Room', 'user_001');
            roomManager.joinRoom(room.roomId, 'user_002');
            roomManager.joinRoom(room.roomId, 'user_003');
            roomManager.joinRoom(room.roomId, 'user_004');
            
            const result = roomManager.joinRoom(room.roomId, 'user_005');
            expect(result).toBe(false);
        });
    });

    describe('leaveRoom', () => {
        it('should remove user from room', () => {
            const room = roomManager.createRoom('Test Room', 'user_001');
            roomManager.joinRoom(room.roomId, 'user_002');
            
            roomManager.leaveRoom(room.roomId, 'user_002');
            
            expect(roomManager.getRoomUsers(room.roomId)).not.toContain('user_002');
        });

        it('should delete room if all users left', () => {
            const room = roomManager.createRoom('Test Room', 'user_001');
            roomManager.leaveRoom(room.roomId, 'user_001');
            
            expect(roomManager.roomExists(room.roomId)).toBe(false);
        });
    });
});

#include "session/ice_handler.h"

class ICEHandlerTest : public ::testing::Test {
protected:
    ICEHandler handler;
};

TEST_F(ICEHandlerTest, ParseCandidateExtractsFields) {
    std::string candidateStr = "candidate:1 1 UDP 2122260223 192.168.1.100 54321 typ host";
    
    ICECandidate candidate = handler.parseCandidate(candidateStr);
    
    EXPECT_EQ(candidate.foundation, "1");
    EXPECT_EQ(candidate.component, 1);
    EXPECT_EQ(candidate.protocol, "UDP");
    EXPECT_EQ(candidate.ip, "192.168.1.100");
    EXPECT_EQ(candidate.port, 54321);
    EXPECT_EQ(candidate.type, "host");
}

TEST_F(ICEHandlerTest, CandidatePriorityOrdering) {
    std::vector<ICECandidate> candidates = {
        handler.parseCandidate("candidate:1 1 UDP 2122260223 192.168.1.100 54321 typ host"),
        handler.parseCandidate("candidate:2 1 UDP 1686052607 203.0.113.1 54322 typ srflx"),
        handler.parseCandidate("candidate:3 1 UDP 92131455 203.0.113.1 54323 typ relay")
    };
    
    std::sort(candidates.begin(), candidates.end(), 
              [](const ICECandidate& a, const ICECandidate& b) {
                  return a.priority > b.priority;
              });
    
    // host > srflx > relay
    EXPECT_EQ(candidates[0].type, "host");
    EXPECT_EQ(candidates[1].type, "srflx");
    EXPECT_EQ(candidates[2].type, "relay");
}

const WebSocket = require('ws');
const { SignalingServer } = require('../../src/server');

describe('Signaling Integration Tests', () => {
    let server;
    let clientA, clientB;
    const PORT = 18080;

    beforeAll((done) => {
        server = new SignalingServer(PORT);
        server.start();
        done();
    });

    afterAll((done) => {
        server.stop();
        done();
    });

    beforeEach((done) => {
        let connected = 0;
        const checkDone = () => {
            connected++;
            if (connected === 2) done();
        };

        clientA = new WebSocket(`ws://localhost:${PORT}`);
        clientA.on('open', checkDone);

        clientB = new WebSocket(`ws://localhost:${PORT}`);
        clientB.on('open', checkDone);
    });

    afterEach(() => {
        clientA.close();
        clientB.close();
    });

    test('Complete signaling flow for 1v1 call', (done) => {
        const messagesA = [];
        const messagesB = [];

        clientA.on('message', (data) => {
            messagesA.push(JSON.parse(data));
        });

        clientB.on('message', (data) => {
            const msg = JSON.parse(data);
            messagesB.push(msg);

            // 当收到 offer 时发送 answer
            if (msg.type === 'offer') {
                clientB.send(JSON.stringify({
                    type: 'answer',
                    from: 'user_b',
                    to: 'user_a',
                    roomId: msg.roomId,
                    sdp: 'fake_answer_sdp'
                }));
            }
        });

        // A 登录
        clientA.send(JSON.stringify({ type: 'login', userId: 'user_a' }));

        setTimeout(() => {
            // A 创建房间
            clientA.send(JSON.stringify({ type: 'createRoom', userId: 'user_a', roomName: 'Test' }));
        }, 100);

        setTimeout(() => {
            // B 登录
            clientB.send(JSON.stringify({ type: 'login', userId: 'user_b' }));
        }, 200);

        setTimeout(() => {
            // B 加入房间
            const roomCreated = messagesA.find(m => m.type === 'roomCreated');
            expect(roomCreated).toBeDefined();
            
            clientB.send(JSON.stringify({ 
                type: 'joinRoom', 
                userId: 'user_b', 
                roomId: roomCreated.roomId 
            }));
        }, 300);

        setTimeout(() => {
            // A 发送 offer
            const joinSuccess = messagesB.find(m => m.type === 'joinRoomSuccess');
            if (joinSuccess) {
                clientA.send(JSON.stringify({
                    type: 'offer',
                    from: 'user_a',
                    to: 'user_b',
                    roomId: joinSuccess.roomId,
                    sdp: 'fake_offer_sdp'
                }));
            }
        }, 400);

        setTimeout(() => {
            // 验证完整流程
            expect(messagesA.find(m => m.type === 'loginSuccess')).toBeDefined();
            expect(messagesA.find(m => m.type === 'roomCreated')).toBeDefined();
            expect(messagesA.find(m => m.type === 'userJoined')).toBeDefined();
            expect(messagesA.find(m => m.type === 'answer')).toBeDefined();
            
            expect(messagesB.find(m => m.type === 'loginSuccess')).toBeDefined();
            expect(messagesB.find(m => m.type === 'joinRoomSuccess')).toBeDefined();
            expect(messagesB.find(m => m.type === 'offer')).toBeDefined();
            
            done();
        }, 600);
    });

class WebRTCConnectionTest : public ::testing::Test {
protected:
    std::unique_ptr<SessionManager> managerA;
    std::unique_ptr<SessionManager> managerB;
    
    void SetUp() override {
        managerA = std::make_unique<SessionManager>();
        managerB = std::make_unique<SessionManager>();
    }
};

TEST_F(WebRTCConnectionTest, EstablishP2PConnection) {
    std::promise<void> connectedPromise;
    auto connectedFuture = connectedPromise.get_future();

    // 设置连接状态回调
    managerB->setOnConnectionStateChange([&](const std::string& peerId, 
                                              webrtc::PeerConnectionInterface::IceConnectionState state) {
        if (state == webrtc::PeerConnectionInterface::kIceConnectionConnected) {
            connectedPromise.set_value();
        }
    });

    // 模拟 SDP 交换
    managerA->initialize("ws://localhost:8080");
    managerB->initialize("ws://localhost:8080");

    // 创建 offer
    std::string offer = managerA->createOffer("peer_b");
    managerB->setRemoteDescription("peer_a", offer, "offer");
    
    // 创建 answer
    std::string answer = managerB->createAnswer("peer_a");
    managerA->setRemoteDescription("peer_b", answer, "answer");

    // 等待连接建立（超时 10 秒）
    auto status = connectedFuture.wait_for(std::chrono::seconds(10));
    EXPECT_EQ(status, std::future_status::ready);
}

// tests/performance/latency_test.js
const { performance } = require('perf_hooks');

describe('Latency Tests', () => {
    test('Audio latency should be under 300ms', async () => {
        const iterations = 100;
        const latencies = [];

        for (let i = 0; i < iterations; i++) {
            const start = performance.now();
            
            // 发送音频包并等待回声
            await sendAudioPacketAndWaitForEcho();
            
            const end = performance.now();
            latencies.push(end - start);
        }

        const avgLatency = latencies.reduce((a, b) => a + b) / iterations;
        const maxLatency = Math.max(...latencies);
        const p95Latency = latencies.sort((a, b) => a - b)[Math.floor(iterations * 0.95)];

        console.log(`Average latency: ${avgLatency.toFixed(2)}ms`);
        console.log(`Max latency: ${maxLatency.toFixed(2)}ms`);
        console.log(`P95 latency: ${p95Latency.toFixed(2)}ms`);

        expect(avgLatency).toBeLessThan(300);
        expect(p95Latency).toBeLessThan(500);
    });
});

// tests/performance/concurrent_test.js
describe('Concurrent Users Tests', () => {
    test('Server should handle 100 concurrent users', async () => {
        const clients = [];
        const promises = [];

        // 创建 100 个并发连接
        for (let i = 0; i < 100; i++) {
            const client = new WebSocket('ws://localhost:8080');
            clients.push(client);

            promises.push(new Promise((resolve, reject) => {
                client.on('open', () => {
                    client.send(JSON.stringify({
                        type: 'login',
                        userId: `user_${i}`
                    }));
                });
                client.on('message', (data) => {
                    const msg = JSON.parse(data);
                    if (msg.type === 'loginSuccess') {
                        resolve();
                    }
                });
                client.on('error', reject);
                
                // 超时
                setTimeout(() => reject(new Error('Timeout')), 10000);
            }));
        }

        const results = await Promise.allSettled(promises);
        const fulfilled = results.filter(r => r.status === 'fulfilled').length;

        // 清理
        clients.forEach(c => c.close());

        expect(fulfilled).toBeGreaterThanOrEqual(95); // 95% 成功率
    });
});
