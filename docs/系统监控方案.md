### 1.2 监控目标

| 监控维度 | 监控对象 | 关键指标 |
|----------|----------|----------|
| **基础设施** | 服务器、网络 | CPU、内存、磁盘、网络流量 |
| **应用服务** | 信令服务器、TURN | 进程状态、请求量、响应时间 |
| **业务指标** | 会议、用户 | 在线用户数、会议数量、通话质量 |
| **WebRTC** | 连接、媒体 | ICE 状态、码率、丢包率、延迟 |

### 2.3 WebRTC 质量指标

```javascript
// WebRTC 统计指标
const webrtcMetrics = {
    // 连接指标
    iceConnectionState: new promClient.Gauge({
        name: 'webrtc_ice_connection_state',
        help: 'ICE 连接状态 (0=new, 1=checking, 2=connected, 3=completed, 4=failed)',
        labelNames: ['server', 'session']
    }),
    
    // 媒体指标
    videoBitrate: new promClient.Gauge({
        name: 'webrtc_video_bitrate_kbps',
        help: '视频码率 (kbps)',
        labelNames: ['server', 'session', 'direction']
    }),
    
    audioBitrate: new promClient.Gauge({
        name: 'webrtc_audio_bitrate_kbps',
        help: '音频码率 (kbps)',
        labelNames: ['server', 'session', 'direction']
    }),
    
    // 质量指标
    packetLoss: new promClient.Gauge({
        name: 'webrtc_packet_loss_percent',
        help: '丢包率百分比',
        labelNames: ['server', 'session', 'media_type']
    }),
    
    rtt: new promClient.Gauge({
        name: 'webrtc_rtt_ms',
        help: '往返延迟 (ms)',
        labelNames: ['server', 'session']
    }),
    
    jitter: new promClient.Gauge({
        name: 'webrtc_jitter_ms',
        help: '抖动 (ms)',
        labelNames: ['server', 'session', 'media_type']
    }),
    
    // 帧率
    frameRate: new promClient.Gauge({
        name: 'webrtc_frame_rate_fps',
        help: '视频帧率',
        labelNames: ['server', 'session', 'direction']
    })
};

// 从 WebRTC Stats 采集指标
async function collectWebRTCStats(pc, sessionId) {
    const stats = await pc.getStats();
    
    stats.forEach(report => {
        if (report.type === 'outbound-rtp') {
            const bitrate = calculateBitrate(report.bytesSent, report.timestamp);
            
            if (report.kind === 'video') {
                webrtcMetrics.videoBitrate.set(
                    { server: hostname, session: sessionId, direction: 'outbound' },
                    bitrate
                );
                webrtcMetrics.frameRate.set(
                    { server: hostname, session: sessionId, direction: 'outbound' },
                    report.framesPerSecond
                );
            } else {
                webrtcMetrics.audioBitrate.set(
                    { server: hostname, session: sessionId, direction: 'outbound' },
                    bitrate
                );
            }
        }
        
        if (report.type === 'inbound-rtp') {
            const lossRate = report.packetsLost / (report.packetsReceived + report.packetsLost);
            
            webrtcMetrics.packetLoss.set(
                { server: hostname, session: sessionId, media_type: report.kind },
                lossRate * 100
            );
            
            webrtcMetrics.jitter.set(
                { server: hostname, session: sessionId, media_type: report.kind },
                report.jitter * 1000  // 转换为毫秒
            );
        }
        
        if (report.type === 'candidate-pair' && report.state === 'succeeded') {
            webrtcMetrics.rtt.set(
                { server: hostname, session: sessionId },
                report.currentRoundTripTime * 1000
            );
        }
    });
}
```

### 3.3 日志类型

| 日志类型 | 内容 | 级别 | 保留周期 |
|----------|------|------|----------|
| **访问日志** | 连接、加入、离开 | INFO | 7 天 |
| **业务日志** | 会议创建、消息转发 | INFO | 30 天 |
| **错误日志** | 异常、错误 | ERROR | 90 天 |
| **安全日志** | 认证、授权事件 | WARN | 90 天 |
| **性能日志** | 响应时间、资源使用 | DEBUG | 7 天 |

## 4. 告警配置

### 4.1 告警规则定义

```yaml
# prometheus/alert-rules.yml

groups:
  - name: infrastructure
    rules:
      # CPU 告警
      - alert: HighCPUUsage
        expr: node_cpu_usage > 80
        for: 5m
        labels:
          severity: warning
        annotations:
          summary: "高 CPU 使用率"
          description: "服务器 {{ $labels.server }} CPU 使用率 {{ $value }}%"
      
      - alert: CriticalCPUUsage
        expr: node_cpu_usage > 95
        for: 2m
        labels:
          severity: critical
        annotations:
          summary: "CPU 使用率危急"
          description: "服务器 {{ $labels.server }} CPU 使用率 {{ $value }}%"
      
      # 内存告警
      - alert: HighMemoryUsage
        expr: node_memory_usage > 85
        for: 5m
        labels:
          severity: warning
        annotations:
          summary: "高内存使用率"
          description: "服务器 {{ $labels.server }} 内存使用率 {{ $value }}%"
      
      # 磁盘告警
      - alert: DiskSpaceLow
        expr: node_disk_usage > 85
        for: 5m
        labels:
          severity: warning
        annotations:
          summary: "磁盘空间不足"
          description: "服务器 {{ $labels.server }} 磁盘 {{ $labels.mount }} 使用率 {{ $value }}%"

  - name: application
    rules:
      # 连接数告警
      - alert: HighConnectionCount
        expr: signaling_ws_connections > 1000
        for: 5m
        labels:
          severity: warning
        annotations:
          summary: "连接数过高"
          description: "服务器 {{ $labels.server }} WebSocket 连接数 {{ $value }}"
      
      # 错误率告警
      - alert: HighErrorRate
        expr: rate(signaling_errors_total[5m]) > 10
        for: 2m
        labels:
          severity: warning
        annotations:
          summary: "错误率过高"
          description: "服务器 {{ $labels.server }} 错误类型 {{ $labels.type }} 速率 {{ $value }}/s"
      
      # 服务不可用
      - alert: ServiceDown
        expr: up{job="signaling"} == 0
        for: 1m
        labels:
          severity: critical
        annotations:
          summary: "服务不可用"
          description: "服务 {{ $labels.instance }} 已停止"

  - name: webrtc
    rules:
      # 丢包告警
      - alert: HighPacketLoss
        expr: webrtc_packet_loss_percent > 10
        for: 2m
        labels:
          severity: warning
        annotations:
          summary: "高丢包率"
          description: "会话 {{ $labels.session }} 丢包率 {{ $value }}%"
      
      # 延迟告警
      - alert: HighLatency
        expr: webrtc_rtt_ms > 300
        for: 2m
        labels:
          severity: warning
        annotations:
          summary: "高网络延迟"
          description: "会话 {{ $labels.session }} RTT {{ $value }}ms"
      
      # ICE 失败
      - alert: ICEConnectionFailed
        expr: webrtc_ice_connection_state == 4
        for: 1m
        labels:
          severity: warning
        annotations:
          summary: "ICE 连接失败"
          description: "会话 {{ $labels.session }} ICE 连接失败"
```

### 4.2 AlertManager 配置

```yaml
# alertmanager/alertmanager.yml

global:
  resolve_timeout: 5m
  smtp_smarthost: 'smtp.example.com:587'
  smtp_from: 'alert@minimeeting.com'
  smtp_auth_username: 'alert@minimeeting.com'
  smtp_auth_password: 'password'

route:
  group_by: ['alertname', 'severity']
  group_wait: 30s
  group_interval: 5m
  repeat_interval: 4h
  receiver: 'default'
  routes:
    - match:
        severity: critical
      receiver: 'critical'
    - match:
        severity: warning
      receiver: 'warning'

receivers:
  - name: 'default'
    email_configs:
      - to: 'team@minimeeting.com'
        send_resolved: true
  
  - name: 'critical'
    email_configs:
      - to: 'oncall@minimeeting.com'
        send_resolved: true
    webhook_configs:
      - url: 'https://hooks.slack.com/services/xxx'
        send_resolved: true
    pagerduty_configs:
      - service_key: 'xxx'
        severity: critical
  
  - name: 'warning'
    email_configs:
      - to: 'team@minimeeting.com'
        send_resolved: true

inhibit_rules:
  - source_match:
      severity: 'critical'
    target_match:
      severity: 'warning'
    equal: ['alertname', 'server']
```

```cpp
// Qt 客户端指标采集
class MetricsCollector : public QObject {
    Q_OBJECT
    
public:
    void start() {
        m_timer.start(10000);  // 每 10 秒上报一次
        connect(&m_timer, &QTimer::timeout, this, &MetricsCollector::collect);
    }
    
    void collect() {
        QJsonObject metrics;
        
        // 系统指标
        metrics["cpu_usage"] = getCpuUsage();
        metrics["memory_usage"] = getMemoryUsage();
        
        // WebRTC 指标
        if (m_peerConnection) {
            QJsonObject webrtc = collectWebRTCStats();
            metrics["webrtc"] = webrtc;
        }
        
        // 应用指标
        metrics["app_uptime"] = m_uptime.elapsed() / 1000;
        metrics["app_version"] = APP_VERSION;
        
        // 上报到服务器
        uploadMetrics(metrics);
    }
    
private:
    QJsonObject collectWebRTCStats() {
        QJsonObject stats;
        
        // 获取 WebRTC 统计
        auto rtcStats = m_peerConnection->GetStats();
        
        for (const auto& report : rtcStats) {
            if (report.type == "outbound-rtp") {
                stats["bytes_sent"] = report.bytesSent;
                stats["packets_sent"] = report.packetsSent;
            }
            if (report.type == "inbound-rtp") {
                stats["bytes_received"] = report.bytesReceived;
                stats["packets_lost"] = report.packetsLost;
            }
            if (report.type == "candidate-pair") {
                stats["rtt"] = report.currentRoundTripTime * 1000;
            }
        }
        
        return stats;
    }
    
    void uploadMetrics(const QJsonObject& metrics) {
        QNetworkRequest request(QUrl(m_metricsUrl));
        request.setHeader(QNetworkRequest::ContentTypeHeader, "application/json");
        
        QJsonDocument doc(metrics);
        m_networkManager->post(request, doc.toJson());
    }
    
    QTimer m_timer;
    QNetworkAccessManager* m_networkManager;
    webrtc::PeerConnectionInterface* m_peerConnection;
    QElapsedTimer m_uptime;
    QString m_metricsUrl = "https://metrics.minimeeting.com/api/v1/metrics";
};
```

### 6.2 指标聚合服务

```javascript
// 指标聚合服务
class MetricsAggregator {
    constructor() {
        this.metrics = new Map();
        this.aggregationWindow = 60000;  // 1 分钟
    }
    
    // 接收指标
    receive(clientId, metrics) {
        const timestamp = Date.now();
        const key = `${clientId}:${Math.floor(timestamp / this.aggregationWindow)}`;
        
        if (!this.metrics.has(key)) {
            this.metrics.set(key, {
                count: 0,
                sum: {},
                min: {},
                max: {}
            });
        }
        
        const agg = this.metrics.get(key);
        agg.count++;
        
        // 聚合数值指标
        for (const [name, value] of Object.entries(metrics)) {
            if (typeof value === 'number') {
                agg.sum[name] = (agg.sum[name] || 0) + value;
                agg.min[name] = Math.min(agg.min[name] || Infinity, value);
                agg.max[name] = Math.max(agg.max[name] || -Infinity, value);
            }
        }
    }
    
    // 计算聚合结果
    getAggregated(windowStart, windowEnd) {
        const results = [];
        
        for (const [key, agg] of this.metrics) {
            const [clientId, window] = key.split(':');
            const timestamp = window * this.aggregationWindow;
            
            if (timestamp >= windowStart && timestamp < windowEnd) {
                results.push({
                    clientId,
                    timestamp,
                    count: agg.count,
                    avg: this.calculateAvg(agg),
                    min: agg.min,
                    max: agg.max
                });
            }
        }
        
        return results;
    }
    
    calculateAvg(agg) {
        const avg = {};
        for (const [name, sum] of Object.entries(agg.sum)) {
            avg[name] = sum / agg.count;
        }
        return avg;
    }
}
```
